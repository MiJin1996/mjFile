📌전체 코드 흐름
    public class Hero {

        private String name;
        private int hp;

        @Override
        public String toString() { //🚨 객체 정보를 문자열로 반환
            return "Hero [name=" + name + ", hp=" + hp + "]";
        }

        @Override
        public boolean equals(Object obj) { //🚨 객체의 내용(값)이 같은지 비교
            if (this == obj) return true;
            if (!(obj instanceof Hero)) return false;

            Hero other = (Hero) obj;

            return hp == other.hp &&
                   Objects.equals(name, other.name);
        }

        @Override
        public int hashCode() { //🚨 객체의 해시값 생성 (equals()와 함께 사용
            return Objects.hash(name, hp);
        }
    }
